# llama-cpp-python — local-LLM Arm 2 (IMPROVEMENT_PLAN §9), CPU-only.
# Install the prebuilt CPU wheel (no MSVC/CMake build) with:
#   pip install "llama-cpp-python==0.3.2" --only-binary=:all: \
#       --extra-index-url https://abetlen.github.io/llama-cpp-python/whl/cpu
# Pinned to 0.3.2: the newest CPU wheels (e.g. 0.3.25) are built requiring
# AVX-512 and fault with STATUS_ILLEGAL_INSTRUCTION (0xc000001d) on CPUs that
# lack it (e.g. AMD Zen 2 / Ryzen 5000U, which has AVX2 but not AVX-512).
# 0.3.2 runs on an AVX2 baseline. This is itself a deployability finding — and
# the reason Arm 2 is not deployed to the free HF Space.
llama-cpp-python==0.3.2
